
FROM node:18-alpine as build
WORKDIR /app
# Expects package_json.txt to be renamed to package.json
COPY package.json ./
RUN npm install
COPY . .
RUN npm run build

FROM nginx:stable-alpine
# Expects backend/nginx_conf.txt to be renamed to backend/nginx.conf
COPY --from=build /app/dist /usr/share/nginx/html
COPY backend/nginx.conf /etc/nginx/conf.d/default.conf
EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]
